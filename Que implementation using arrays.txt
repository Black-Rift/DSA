#include <iostream>
using namespace std;
#define n 20
class que{
    int* arr;
    int front;
    int back;
    public:
    que(){
        arr=new int[n];
        front=-1;
        back=-1;
    }
    void push(int x){
        if(back==n-1){
            cout<<"Que Overflow";
            return;
        }
        back++;
        arr[back]=x;
        if(front==-1){
            front++;
        }
    }
    void pop(){
        if(front==-1 || front>back){
            cout<<"Que Underfolw";
            return;
        }
        front++;
    }
    int peek(){
         if(front==-1 || front>back){
            cout<<"No elements in Que";
            return -1;
        }
        return arr[front];
    }
    bool isempty(){
         if(front==-1 || front>back){
            return true;
        }
        return false;
    }
};
