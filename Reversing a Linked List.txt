// ITERATIVE METHOD
node* reverse(node* &head){
  node* prevptr=NULL;
  node* currptr=head;
  node* nextptr;
  while(currptr!=NULL){
    nextptr=currptr->next;
    currptr->next=prevptr;
    prevptr=currptr;
    currptr=nextptr;
    }
  return prevptr;
  }

// RECURSIVE METHOD
  node* recrev(node* &head){
    node* newhead=recrev(head->next);
